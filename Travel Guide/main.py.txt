import openai
import gradio as gr

# Set your OpenAI GPT-3 API key
openai.api_key = 'YOUR_API_KEY'

# Define the Travel Guide function
def travel_guide(destination, num_days):
    # Customize the prompt based on the destination and number of days
    prompt_text = f"You: I am planning to visit {destination} for {num_days} days. Can you provide me with some travel recommendations?"

    # Make a request to the OpenAI API
    response = openai.Completion.create(
        engine="text-davinci-003",  # Choose the appropriate engine
        prompt=prompt_text,
        max_tokens=150  # Adjust the max tokens as needed
    )

    # Extract and return the model's travel recommendations
    reply = response['choices'][0]['text'].strip()
    return reply

# Create a Gradio interface
iface = gr.Interface(
    fn=travel_guide,
    inputs=["text", "text"],
    outputs="text",
    live=True,
    title="Travel Guide Chatbot",
    description="Ask for travel recommendations by providing a destination and the number of days you plan to stay."
)

# Launch the interface
iface.launch()
